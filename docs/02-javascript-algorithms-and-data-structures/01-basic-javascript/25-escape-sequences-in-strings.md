# Escape Sequences in Strings
## Summary
{% highlight JavaScript %}
\'	single quote
\"	double quote
\\	backslash
\n	newline
\r	carriage return
\t	tab
\b	word boundary
\f	form feed
{% endhighlight %}


## Final Code

{% highlight JavaScript %}
var myStr = "FirstLine\n\t\\SecondLine\nThirdLine"; // Change this line
{% endhighlight %}
